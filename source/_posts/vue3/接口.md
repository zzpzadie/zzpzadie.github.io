# 第一种：get方式
```bash
#1.1 普通get 不需要传参 只有URL和method
export const getNotice = () => {
return axios({
url: ‘/web/v1/web/notice’,
method: ‘get’
})
}
#1.2.1 get需要传参 使用data传参（params: data）
#params 属性用于将 data 对象中的键值对作为 URL 查询参数附加到请求的 URL 上。例如，如果 data 是 {key1: 'value1', key2: 'value2'}，那么请求的 URL 将会是 /user/v1/account/topinfo?key1=value1&key2=value2。
#所以data相当于自命名的一个变量，可以命名为query或任意其他单词
export const getTopInfo = (data) => {
return axios({
url: ‘/user/v1/account/topinfo’,
method: ‘get’,
params: data
})
}
#1.2.2 get需要传参 使用json格式传参（data: data）
#data 属性用于将 data 对象作为请求体发送给服务器。这种情况下，服务器需要处理 JSON 格式的数据。
export const getTopInfo = (data) => {
return axios({
url: ‘/user/v1/account/topinfo’,
method: ‘get’,
data: data
})
}
#或者直接把data: data用data简写，所以data相当于自命名的一个变量，可以命名为query、params或任意其他单词，例：
export const getTopInfo = (params) => {
return axios({
url: ‘/user/v1/account/topinfo’,
method: ‘get’,
params
})
}
#1.3.1 get需要传参 使用url传参
#将 data 对象对作为 URL 查询参数附加到请求的 URL 上。例如，如果 data 是 {key1: 'value1', key2: 'value2'}，那么请求的 URL 将会是/web/v4/guildTask/autoAssign/{"key1":"value1","key2":"value2"}

export const deleteAssign = function (data) {
return axios({
#  // url传参data或data.autoAssignId
url: ‘/web/v4/guildTask/autoAssign/’ + data, 
method: ‘get’
})
}
#1.3.2 get需要传参 使用url传参
export const deleteAssign = function (data) {
return axios({
url: ‘/web/v4/guildTask/autoAssign?autoAssignId=’ + data.autoAssignId, // url传参
method: ‘get’
})
}
#还有多个拼接的，例子：
export function transmit(type,dataId) {
    return request({
        url: '/orbit-elements/tle/sendTle?type=' +type +'&dataId=' +dataId,
        method: 'get'
    })
}

export function listPackage(query) {
  return request({
    url: '/telemetry-config/sourcePackage/list?' + query,
    method: 'get',
  })
}

#1.3.3 get需要传参 使用url传参，$为占位符，如id为123， url为/system/cockpit/123
export function deleteCockpit(id) {
  return request({
    url: `/system/cockpit/${id}`,
    method: 'delete'
  })
}
#也有数组的：
export function planTaskTest(id) {
    return request({
        url: `/taskplan/plan/sendPlan?ids=${id.ids}`,
        method: 'get'
    })
}
#同：把get换成方法类型即可
export function getUser(userId) {
  let emptyUserid = parseStrEmpty(userId)
  return api.get(`/system/user/${emptyUserid}`)
}
```
